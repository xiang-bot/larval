    public function addform(Request $request){
        if ($request->isMethod('get')){
            return view('showform'); //显示添加记录的视图
        }elseif($request->isMethod('post')){
            $all = $request->all();
            if (isset($all['aihao']))
                $all['hobby'] = implode(',',$all['aihao']);
            else
                $all['hobby'] = '';
            $m = new Tb_info();
            $rst = $m->create($all);
            //$rst = Tb_info::create($all);
            dump($rst);
        }
    }
