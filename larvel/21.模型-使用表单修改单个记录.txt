    public function xiugai3(Request $request,$id){
        if ($request->isMethod('get')){
            $m = Tb_info::find($id);
            return view('update3',compact('m'));
        }elseif($request->isMethod('post')){
            $all = $request->all();
            $m = new Tb_info();
            $m2 = $m->find($id);
            $m2->name = $all['name'];
            $m2->sex = $all['sex'];
            $m2->banji = $all['banji'];
            $m2->yuwen  = $all['yuwen'];
            if (isset($all['aihao']))
                $m2->hobby = implode(',',$all['aihao']);
            else
                $m2->hobby = '';
            $rst = $m2->save();
            dump($rst);
        }
    }
