第二步：配置域名：
            hosts.txt中加入：127.0.0.1    fenye.com
            vhosts.conf.txt加入：
	<VirtualHost *:80>
	    DocumentRoot "C:\fenye\public"
	    ServerName fenye.com
 	   ServerAlias 
 	 <Directory "C:\C:\fenye\public\public">
  	    Options FollowSymLinks ExecCGI
  	    AllowOverride All
   	   Order allow,deny
    	  Allow from all
    	 Require all granted
  	</Directory>
	</VirtualHost>   
              重启apache

use App\Tb_info;
use Illuminate\Pagination\Paginator;
use Session;

        public function fenye(Request $request){
        $m = new Tb_info();
        $data = $m->orderby('id')->paginate(4);
        //Paginator::defaultView('vendor.pagination.default');
        $p = $data->currentPage();
        session(['p' => $p]);
        return view('fenye',compact('data','p'));
    }

    public function xiugai(Request $request,$id){
        if ($request->isMethod('get')){
            $m1 = new Tb_info();
            $m =  $m1->find($id);
            return view('modify',compact('m'));
        }elseif($request->isMethod('post')){
            $all = $request->all();
            $m = new Tb_info();
            $m2 = $m->find($id);
            $m2->name = $all['name'];
            $m2->sex = $all['sex'];
            $m2->class = $all['class'];
            $m2->photo  = $all['photo'];
            if (isset($all['aihao']))
                $m2->hobby = implode(',',$all['aihao']);
            else
                $m2->hobby = '';
            $rst = $m2->save();
            $p = session('p');
            return redirect("/fenye?page=$p");
        }
    }

    public function del($id){
        Tb_info::destroy($id);
        $p = session('p');
        return redirect("/fenye?page=$p");
    }
