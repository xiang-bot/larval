1.  public function del1()
    {
        $db = DB::table('tb_info');
        $rst = $db->delete();  //删除全部记录
        dd($rst);              //受影响的记录条数
    }


2.  public function del2()
    {
        $db = DB::table('tb_info');
        $rst = $db->where('name','刘备')->delete();  //删除name为刘备的记录
        //where和orWhere在这里也可以使用，使用方法和前面相同
        dd($rst);              //受影响的记录条数
    }


3.  public function del3()
    {
        $db = DB::table('tb_info');
        $rst = $db->truncate();  //删除全部记录，返回null
        dd($rst);
    }
