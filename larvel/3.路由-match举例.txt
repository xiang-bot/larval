1.路由：
Route::match(['get','post'],'/login',function(Request $request){
    if ($request->isMethod('get')){
        echo $request->input('user')."<br />";
        //return view('login');
    }elseif($request->isMethod('post')){
        echo $request->input('user')."<br />";
        echo $request->input('pwd');
    }
})->name('login');
